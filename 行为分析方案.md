# 宠物行为分析模型方案

> 与品种识别模型并行开发，共用同一套 FastAPI 推理服务

---

## 一、任务定义

| 对比项 | 品种识别（已在训练） | 行为分析（本方案） |
|--------|--------------------|--------------------|
| 问的问题 | 这是什么品种？ | 这只宠物在做什么？ |
| 输入 | 宠物图片 | 宠物图片（静图即可） |
| 输出 | 37 个品种标签 | 6 个行为标签 |
| 模型 | EfficientNet-B4 | MobileNetV3-Large |
| 数据集 | Oxford-IIIT Pet | 混合数据集（见下） |

**识别的 6 种行为：**

| 标签 | 中文 | 典型特征 |
|------|------|----------|
| `sleeping` | 睡觉 | 趴卧、眼闭、蜷缩 |
| `eating` | 进食 | 头低近碗、咀嚼动作 |
| `playing` | 玩耍 | 扑跳、叼玩具、奔跑 |
| `sitting` | 安静坐立 | 端坐、站立、凝视 |
| `grooming` | 梳理毛发 | 舔爪、舔毛、自我清洁 |
| `walking` | 行走 | 正常步态移动 |

---

## 二、数据集方案（推荐混合策略）

### 2.1 数据集选型对比

| 数据集 | 类型 | 规模 | 行为类别 | 获取难度 | 推荐度 |
|--------|------|------|----------|----------|--------|
| **Roboflow Cat Dog Status** | 图像 | 2,291 张 | eating / walking / sleeping 等 | ⭐ 极易（API一键下载） | ★★★★★ |
| **Kaggle Dog Behavior Monitoring** | 图像+标注 | 未知 | 多级姿态+行为 | ⭐⭐ 易 | ★★★★ |
| **Kaggle Animal Behavior Analysis** | 图像 | 未知 | 多种行为 | ⭐⭐ 易 | ★★★ |
| **Animal Kingdom (CVPR 2022)** | 视频 | 30K 视频段 | 140 类，850+ 物种 | ⭐⭐⭐⭐ 难（需申请） | ★★（太大太复杂）|
| **自建爬虫数据集** | 图像 | 自定义 | 完全自定义 | ⭐⭐ 中等 | ★★★★（可补充） |

### 2.2 最终方案：三源混合

```
行为数据集（目标：每类 300-500 张，共约 2000 张）
├── 来源 A：Roboflow Cat Dog Status（~2291 张，直接下载）
│   └── 已有标签：eating / walking / sleeping 等，直接可用
├── 来源 B：Kaggle Dog Behavior Monitoring（补充 sitting / grooming）
│   └── 按行为类别筛选对应图片
└── 来源 C：自动爬虫补充（每类不足时用脚本抓取）
    └── 关键词：cat sleeping / dog eating / cat grooming 等
```

> **为什么不用 Animal Kingdom？**
> 该数据集 30K 视频段 + 140 类，面向视频动作识别，
> 下载需要学术申请，数据量过大，与本项目静图分类方案不匹配。

## 四、数据整理脚本

```python
# 07_prepare_behavior_data.py
"""
将所有来源的行为图片整理为统一的 train/val/test 结构
输出：behavior_dataset/train|val|test/sleeping|eating|playing|...
"""
import shutil, random, json
from pathlib import Path
from PIL import Image
from tqdm import tqdm

SRC_DIRS = [
    Path("./behavior_raw"),          # 爬虫 + Roboflow 原始图
]
DEST_DIR  = Path("./behavior_dataset")
META_FILE = Path("./models/behavior_class_meta.json")

CLASSES     = ["sleeping", "eating", "playing", "sitting", "grooming", "walking"]
SPLIT_RATIO = (0.8, 0.1, 0.1)
MIN_IMAGES  = 100    # 每类最少要求
RANDOM_SEED = 42

def is_valid_image(path: Path) -> bool:
    try:
        Image.open(path).verify()
        return True
    except:
        return False

def main():
    random.seed(RANDOM_SEED)
    for split in ("train", "val", "test"):
        shutil.rmtree(DEST_DIR / split, ignore_errors=True)

    stats = {}
    for cls in CLASSES:
        imgs = []
        for src in SRC_DIRS:
            cls_dir = src / cls
            if cls_dir.exists():
                imgs += [p for p in cls_dir.rglob("*.jpg") if is_valid_image(p)]
                imgs += [p for p in cls_dir.rglob("*.png") if is_valid_image(p)]

        random.shuffle(imgs)
        n = len(imgs)
        if n < MIN_IMAGES:
            print(f"  ⚠ {cls}: 仅 {n} 张，建议补充至 {MIN_IMAGES} 张")

        t_end = int(n * SPLIT_RATIO[0])
        v_end = t_end + int(n * SPLIT_RATIO[1])
        splits = {"train": imgs[:t_end], "val": imgs[t_end:v_end], "test": imgs[v_end:]}

        for split, files in splits.items():
            dest = DEST_DIR / split / cls
            dest.mkdir(parents=True, exist_ok=True)
            for i, f in enumerate(files):
                shutil.copy(f, dest / f"{cls}_{i:04d}{f.suffix}")

        stats[cls] = n
        print(f"  {cls}: {n} 张 → train={t_end} val={v_end-t_end} test={n-v_end}")

    # 保存类别元数据
    META_FILE.parent.mkdir(exist_ok=True)
    class_to_idx = {c: i for i, c in enumerate(CLASSES)}
    BEHAVIOR_CN  = {
        "sleeping": "睡觉", "eating": "进食", "playing": "玩耍",
        "sitting": "安静", "grooming": "梳毛", "walking": "行走"
    }
    with open(META_FILE, "w") as f:
        json.dump({
            "class_to_idx": class_to_idx,
            "idx_to_class": {str(v): k for k, v in class_to_idx.items()},
            "behavior_cn":  BEHAVIOR_CN,
            "num_classes":  len(CLASSES),
        }, f, ensure_ascii=False, indent=2)

    print(f"\n数据集整理完成，元数据：{META_FILE}")
    print("下一步：python 08_train_behavior.py")

if __name__ == "__main__":
    main()
```

## 五、训练脚本

```python
# 08_train_behavior.py
"""
训练行为分析模型：MobileNetV3-Large（比 EfficientNet-B4 轻量，适合行为粗粒度分类）
Apple M 芯片 MPS 加速
"""
import json, time
from pathlib import Path
import torch
import torch.nn as nn
from torch.utils.data import DataLoader
from torchvision import datasets, transforms
import timm
from tqdm import tqdm

BASE_DIR  = Path(__file__).parent
DATA_DIR  = BASE_DIR / "behavior_dataset"
MODEL_DIR = BASE_DIR / "models"
CKPT_PATH = MODEL_DIR / "behavior_classifier_best.pth"
META_FILE = MODEL_DIR / "behavior_class_meta.json"

MODEL_NAME   = "mobilenetv3_large_100"   # 更轻量，推理更快
IMG_SIZE     = 224                        # MobileNet 标准输入
BATCH_SIZE   = 32
PHASE1_LR    = 1e-3
PHASE2_LR    = 5e-5
PHASE1_EPOCH = 5
TOTAL_EPOCH  = 20
NUM_WORKERS  = 0

# 设备
if torch.backends.mps.is_available():
    DEVICE = torch.device("mps")
elif torch.cuda.is_available():
    DEVICE = torch.device("cuda")
else:
    DEVICE = torch.device("cpu")
print(f"使用设备：{DEVICE}")

train_tf = transforms.Compose([
    transforms.Resize((IMG_SIZE + 16, IMG_SIZE + 16)),
    transforms.RandomCrop(IMG_SIZE),
    transforms.RandomHorizontalFlip(),
    transforms.RandomRotation(20),
    transforms.ColorJitter(brightness=0.3, contrast=0.3, saturation=0.2),
    transforms.ToTensor(),
    transforms.Normalize([0.485, 0.456, 0.406], [0.229, 0.224, 0.225]),
])
val_tf = transforms.Compose([
    transforms.Resize((IMG_SIZE, IMG_SIZE)),
    transforms.ToTensor(),
    transforms.Normalize([0.485, 0.456, 0.406], [0.229, 0.224, 0.225]),
])

def run_epoch(model, loader, criterion, optimizer=None):
    training = optimizer is not None
    model.train() if training else model.eval()
    total_loss, correct, total = 0.0, 0, 0
    ctx = torch.enable_grad() if training else torch.no_grad()
    with ctx:
        for imgs, labels in tqdm(loader, leave=False):
            imgs, labels = imgs.to(DEVICE), labels.to(DEVICE)
            logits = model(imgs)
            loss = criterion(logits, labels)
            if training:
                optimizer.zero_grad(); loss.backward(); optimizer.step()
            total_loss += loss.item()
            correct    += (logits.argmax(1) == labels).sum().item()
            total      += labels.size(0)
    return total_loss / len(loader), correct / total

def main():
    MODEL_DIR.mkdir(exist_ok=True)
    train_ds = datasets.ImageFolder(str(DATA_DIR / "train"), transform=train_tf)
    val_ds   = datasets.ImageFolder(str(DATA_DIR / "val"),   transform=val_tf)
    test_ds  = datasets.ImageFolder(str(DATA_DIR / "test"),  transform=val_tf)

    train_ld = DataLoader(train_ds, BATCH_SIZE, shuffle=True,  num_workers=NUM_WORKERS, pin_memory=False)
    val_ld   = DataLoader(val_ds,   BATCH_SIZE, shuffle=False, num_workers=NUM_WORKERS, pin_memory=False)
    test_ld  = DataLoader(test_ds,  BATCH_SIZE, shuffle=False, num_workers=NUM_WORKERS, pin_memory=False)

    print(f"行为类别 {len(train_ds.classes)}：{train_ds.classes}")
    print(f"训练 {len(train_ds)} | 验证 {len(val_ds)} | 测试 {len(test_ds)}")

    model     = timm.create_model(MODEL_NAME, pretrained=True, num_classes=len(train_ds.classes)).to(DEVICE)
    criterion = nn.CrossEntropyLoss(label_smoothing=0.1)

    # 阶段一：只训练分类头
    for name, p in model.named_parameters():
        p.requires_grad = ("classifier" in name or "head" in name)
    optimizer = torch.optim.AdamW(filter(lambda p: p.requires_grad, model.parameters()), lr=PHASE1_LR)
    scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(optimizer, T_max=PHASE1_EPOCH)

    best_acc = 0.0
    for epoch in range(1, TOTAL_EPOCH + 1):
        if epoch == PHASE1_EPOCH + 1:
            for p in model.parameters(): p.requires_grad = True
            optimizer = torch.optim.AdamW(model.parameters(), lr=PHASE2_LR, weight_decay=1e-4)
            scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(optimizer, T_max=TOTAL_EPOCH - PHASE1_EPOCH)
            print(">> 解冻全部参数，全网微调")

        tr_loss, tr_acc = run_epoch(model, train_ld, criterion, optimizer)
        val_loss, val_acc = run_epoch(model, val_ld, criterion)
        scheduler.step()
        print(f"Epoch {epoch:02d}/{TOTAL_EPOCH} | train={tr_acc:.4f} val={val_acc:.4f}")

        if val_acc > best_acc:
            best_acc = val_acc
            torch.save(model.state_dict(), CKPT_PATH)
            print(f"  ✓ 保存最佳模型 val_acc={val_acc:.4f}")

    model.load_state_dict(torch.load(CKPT_PATH, map_location=DEVICE))
    _, test_acc = run_epoch(model, test_ld, criterion)
    print(f"\n最终 Test Accuracy: {test_acc:.4f} ({test_acc*100:.1f}%)")
    print("下一步：python 09_export_behavior_onnx.py")

if __name__ == "__main__":
    main()
```

import json, torch, timm, onnx, onnxruntime as ort, numpy as np
from pathlib import Path

BASE_DIR  = Path(__file__).parent
MODEL_DIR = BASE_DIR / "models"
META_FILE = MODEL_DIR / "behavior_class_meta.json"
CKPT_PATH = MODEL_DIR / "behavior_classifier_best.pth"
ONNX_PATH = MODEL_DIR / "behavior_classifier.onnx"

with open(META_FILE) as f:
    meta = json.load(f)

model = timm.create_model("mobilenetv3_large_100", pretrained=False, num_classes=meta["num_classes"])
model.load_state_dict(torch.load(CKPT_PATH, map_location="cpu"))
model.eval()

dummy = torch.randn(1, 3, 224, 224)
torch.onnx.export(model, dummy, str(ONNX_PATH),
    input_names=["image"], output_names=["logits"],
    dynamic_axes={"image": {0: "batch"}, "logits": {0: "batch"}},
    opset_version=17)

onnx.checker.check_model(onnx.load(str(ONNX_PATH)))
print(f"✓ ONNX 导出完成：{ONNX_PATH}")
